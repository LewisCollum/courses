import numpy
import matplotlib.pyplot as pyplot

R = numpy.arange(0.18, 0.32, 0.2)
X = numpy.empty((30, 3))

X[0] = numpy.asarray([100, 100, 100])

table = {}

for rIndex, r in enumerate(R):
    for i in range(1, len(X)):
        A = numpy.asarray([
            [0, 0, 0.33],
            [r, 0, 0],
            [0, 0.71, 0.94]])
        
        X[i] = A.dot(X[i-1])
    table[r] = X

print(table[0.18])
pyplot.plot(numpy.arange(2000, 2030), table[0.18])
pyplot.show()
